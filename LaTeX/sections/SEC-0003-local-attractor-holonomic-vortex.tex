\section{Local Attractor Hierarchy and Nonlocal Holonomic Vortices}
\label{sec:local-attractor-holonomic-vortex}

\subsection{Local attractor hierarchy}
For each bounded or stabilized object $O_i$, postulate a local attractor $A_i$ organizing the local relational flow.

\subsection{Rotating order parameter}
In a neighborhood of $A_i$, define
\begin{equation}
\Psi_i(x)=\sqrt{\rho_i(x)}e^{i\theta_i(x)}.
\end{equation}
If a nonzero angular velocity $\Omega_i$ exists, define the normalized axis
\begin{equation}
n_i = \frac{\Omega_i}{\|\Omega_i\|},
\end{equation}
with poles $P_i^{\pm}=\pm n_i$ and equator
\begin{equation}
E_i = \{x\in S^2 : x\cdot n_i = 0\}.
\end{equation}

\subsection{Coriolis term}
In the rotating frame the local flow acquires the standard Coriolis-like term
\begin{equation}
F_{C,i} = -2\Omega_i \times u_i.
\end{equation}

\subsection{Nonlocal holonomic vortex}
Define the spacetime phase-flow one-form
\begin{equation}
u^{(i)}_\mu = \partial_\mu \theta_i - A^{(i)}_\mu.
\end{equation}
For a closed spacetime loop $\mathcal C$,
\begin{equation}
\Gamma_i[\mathcal C] = \oint_{\mathcal C} u^{(i)}_\mu dx^\mu.
\end{equation}
A nonlocal holonomic vortex is present when
\begin{equation}
\Gamma_i[\mathcal C] \neq 0.
\end{equation}
