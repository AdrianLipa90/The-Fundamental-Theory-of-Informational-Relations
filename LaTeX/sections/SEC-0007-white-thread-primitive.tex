\section{White-Thread U(1) Primitive}
\label{sec:white-thread-primitive}

\subsection{Minimal transport operator}
Define the abelian transport operator along a path $\gamma_{ij}$ by
\begin{equation}
U[\gamma_{ij}] = \mathcal P\exp\!\left(i\int_{\gamma_{ij}} A\right).
\end{equation}
In the U(1) toy sector path ordering is trivial and one obtains
\begin{equation}
U[\gamma_{ij}] = \exp\!\left(i\int_{\gamma_{ij}} A\right).
\end{equation}
This is the abelian Wilson-line analogue of holonomy transport.

\subsection{White-Thread primitive}
Define
\begin{equation}
W_{ij}[\gamma] = \langle \Psi_i | U[\gamma_{ij}] | \Psi_j \rangle.
\end{equation}
For normalized states and unitary transport,
\begin{equation}
|W_{ij}[\gamma]|\le 1.
\end{equation}

\subsection{Status}
This section records the minimal abelian White-Thread primitive only. Later effective pairwise quantities may include additional dressing or calibration layers.
