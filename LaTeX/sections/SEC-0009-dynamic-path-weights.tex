\section{Dynamic Path Weights from Semantic Action}
\label{sec:dynamic-path-weights}

\subsection{Semantic action}
For each candidate path $\gamma^{(a)}$, define
\begin{equation}
\mathcal S(\gamma^{(a)}) = \alpha L(\gamma^{(a)}) + \beta\,\Delta\phi(\gamma^{(a)}) + \kappa D_{\mathrm{rel}}(\gamma^{(a)}) + \mu\,\Pi_{\mathrm{truth}}(\gamma^{(a)}).
\end{equation}

\subsection{Dynamic weights}
For $\sigma\ge 0$, define the normalized weights
\begin{equation}
\alpha_a = \frac{e^{-\sigma \mathcal S(\gamma^{(a)})}}{\sum_b e^{-\sigma \mathcal S(\gamma^{(b)})}}.
\end{equation}
These weights are nonnegative, normalized, and favor lower-action paths when $\sigma>0$.

\subsection{Effective White-Thread}
The resulting action-driven effective amplitude is
\begin{equation}
W_{ij}^{\mathrm{eff}} = \sum_a \alpha_a W_{ij}^{(a)}.
\end{equation}

\subsection{Status}
This section records the first dynamic weighting law for loose-thread bundles. It does not yet derive the final runtime measurement operators or coefficient values.
