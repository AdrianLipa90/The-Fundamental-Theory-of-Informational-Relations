\section{Object-State Energy Functional and Topological Seed Phase}
\label{sec:object-state-energy}

\subsection{Raw energy and normalization}
Define the raw energy of object $i$ by
\begin{equation}
E_i^{raw}=F(\rho_i,p_i,w_i,\nu_i,\vec s_i,\chi_i,\mathcal N_i).
\end{equation}
In the first linear ansatz,
\begin{equation}
E_i^{raw}=\alpha_1\rho_i+\alpha_2 p_i+\alpha_3 w_i+\alpha_4\nu_i+\alpha_5\|\vec s_i\|_2+\alpha_6\delta(\chi_i)+\alpha_7\|\mathcal N_i\|.
\end{equation}
Normalize by
\begin{equation}
\widetilde E_i = \frac{E_i^{raw}}{\sum_j E_j^{raw}},
\qquad
A_i = \sqrt{\widetilde E_i}.
\end{equation}

\subsection{Topological seed phase}
Let
\begin{equation}
s_i = H(\text{content}_i,\text{path}_i,\rho_i,p_i,w_i,\nu_i,\vec s_i,\chi_i,\mathcal N_i),
\end{equation}
and map it to phase by
\begin{equation}
\phi_i = 2\pi\,\mathrm{frac}(g(s_i)).
\end{equation}
The local object state is then
\begin{equation}
\psi_i=A_i e^{i\phi_i}.
\end{equation}
