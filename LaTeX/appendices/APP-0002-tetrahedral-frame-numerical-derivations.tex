\appendix
\section{Numerical Appendix: Tetrahedral Relational Frame on $S^2$}
\label{app:tetrahedral-relational-frame}

\subsection{Canonical vertex realization}
A standard realization of a regular tetrahedron on the unit sphere is obtained from the four sign patterns
\begin{equation}
(1,1,1),\quad (1,-1,-1),\quad (-1,1,-1),\quad (-1,-1,1)
\end{equation}
rescaled by $1/\sqrt{3}$. Thus
\begin{equation}
v_0 = \frac{1}{\sqrt{3}}(1,1,1),\quad
v_1 = \frac{1}{\sqrt{3}}(1,-1,-1),\quad
v_2 = \frac{1}{\sqrt{3}}(-1,1,-1),\quad
v_3 = \frac{1}{\sqrt{3}}(-1,-1,1).
\end{equation}

\subsection{Unit norm check}
Each vertex satisfies
\begin{equation}
\|v_i\|^2 = \frac{1+1+1}{3}=1.
\end{equation}

\subsection{Off-diagonal inner product}
For example,
\begin{equation}
v_0\cdot v_1 = \frac{1-1-1}{3} = -\frac13.
\end{equation}
By symmetry the same holds for all $i\neq j$.

\subsection{Zero centroid}
Summing the four vertices componentwise gives
\begin{equation}
\sum_{i=0}^3 v_i = 0.
\end{equation}

\subsection{Relative phase triple}
Given phases $(\phi_0,\phi_1,\phi_2,\phi_3)$ on the tetrahedral vertices, quotienting by global phase yields the relative triple
\begin{equation}
\gamma_1 = \phi_1-\phi_0,
\qquad
\gamma_2 = \phi_2-\phi_0,
\qquad
\gamma_3 = \phi_3-\phi_0.
\end{equation}
Under a global shift $\phi_i \mapsto \phi_i + \chi$, these differences are invariant.
