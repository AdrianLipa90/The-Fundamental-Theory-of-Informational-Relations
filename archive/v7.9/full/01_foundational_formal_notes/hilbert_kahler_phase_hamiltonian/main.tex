\begin{abstract}
This note rebuilds the phase-first Hamiltonian from the ground up.  The starting point is a two-level Hilbert space whose projective space is the Kahler manifold $\mathbb{CP}^{1}$, equivalently the Bloch sphere with the Fubini--Study metric and Berry curvature.  The standard geometric layer is then extended by model-specific axioms: an Euler/spin constraint selecting a spin-$1/2$ double-valued sector, zeta-zero labels at the spin poles, a tetrahedral depth frame inside the Bloch sphere, a Poincare-disk interference surface, and a Collatz rhythm seeded by twin primes.  The resulting Hamiltonian is not split into ``free particle plus interaction''.  Instead, the free part is the phase/intention generator, while the remainder is the covariant relational motion over the Hilbert-Kahler/Bloch/Poincare geometry.
\end{abstract}

\section{Hilbert space, projectivization, and the Bloch sphere}
Let a normalized spinor be written as
\begin{equation}
\ket{\psi(\theta,\phi)}=\cos\frac{\theta}{2}\ket{0}+e^{i\phi}\sin\frac{\theta}{2}\ket{1}.
\end{equation}
The physical ray $[\psi]$ is invariant under a global phase:
\begin{equation}
\ket{\psi}\sim e^{i\alpha}\ket{\psi}.
\end{equation}
Therefore the physical state space is the projective Hilbert space
\begin{equation}
\mathbb P(\mathcal H)=\{\ket{\psi}:\braket{\psi|\psi}=1\}/U(1)\simeq \mathbb{CP}^{1}\simeq S^2.
\end{equation}
The corresponding Bloch vector is
\begin{equation}
\bm n=\mel{\psi}{\bm\sigma}{\psi}=(\sin\theta\cos\phi,\sin\theta\sin\phi,\cos\theta),
\end{equation}
where $\bm\sigma=(\sigma_x,\sigma_y,\sigma_z)$ are the Pauli matrices.

\subsection{Fubini--Study metric}
The Fubini--Study line element is
\begin{equation}
 ds_{\rm FS}^2=\braket{d\psi|d\psi}-|\braket{\psi|d\psi}|^2.
\end{equation}
For the spinor above one obtains
\begin{equation}
\boxed{ds_{\rm FS}^2=\frac14\left(d\theta^2+\sin^2\theta\,d\phi^2\right).}
\end{equation}
Thus the projective qubit space is a round sphere of radius $1/2$ in the Fubini--Study normalization.

\subsection{Kahler form and Berry curvature}
The Berry connection in a standard gauge may be taken as
\begin{equation}
\mathcal A_B=i\braket{\psi|d\psi}=\frac{1-\cos\theta}{2}\,d\phi,
\end{equation}
up to a gauge sign convention.  Its curvature is
\begin{equation}
\boxed{F_B=d\mathcal A_B=\frac12\sin\theta\,d\theta\wedge d\phi.}
\end{equation}
This curvature is also the Kahler form on $\mathbb{CP}^{1}$ in the same normalization:
\begin{equation}
\Omega_{\rm FS}=F_B.
\end{equation}
Integrating over the sphere gives
\begin{equation}
\int_{S^2}F_B=\int_0^{2\pi}\int_0^\pi \frac12\sin\theta\,d\theta d\phi=2\pi,
\end{equation}
so the first Chern number of the tautological line bundle is
\begin{equation}
\boxed{\frac{1}{2\pi}\int_{S^2}F_B=1.}
\end{equation}
This provides the elementary topological unit of Berry phase on the Bloch sphere.

\section{Euler/spin constraint and polarization}
The spinorial sector is represented by the double cover
\begin{equation}
SU(2)\longrightarrow SO(3).
\end{equation}
A rotation by $2\pi$ changes a spin-$1/2$ state by a sign, while a rotation by $4\pi$ returns it to itself:
\begin{equation}
\boxed{\psi(\alpha+2\pi)=-\psi(\alpha),\qquad \psi(\alpha+4\pi)=\psi(\alpha).}
\end{equation}
This is the formal origin of the $720^\circ$ closure rule.  In the present model this is encoded as an Euler/spin constraint:
\begin{equation}
\Phi_E=s\oint_\gamma \omega_E,
\qquad s=\frac12,
\end{equation}
where $\omega_E$ is an Euler/spin connection along a closed path $\gamma$.

Because $s=1/2$ has two local signs, the state naturally carries a plus/minus polarization:
\begin{equation}
\mathcal H=\mathcal H_+\oplus\mathcal H_-.
\end{equation}
The poles of the Bloch sphere are then distinguished as the eigenstates of $\sigma_z$:
\begin{equation}
N\equiv\ket{0},\qquad S\equiv\ket{1},
\end{equation}
with
\begin{equation}
\sigma_z\ket{0}=+\ket{0},\qquad \sigma_z\ket{1}=-\ket{1}.
\end{equation}
This gives the model its polar axis.

\begin{equation}

\begin{equation}
\boxed{D_\Lambda[\Sigma]=\frac{1}{2\pi}\int_\Sigma \Lambda\,dA_{\rm phys}.}
\end{equation}
For constant $\Lambda$ and physical area $A_\Sigma$ this reduces to
\begin{equation}
\boxed{D_\Lambda=\frac{\Lambda A_\Sigma}{2\pi}.}
\end{equation}
The neutrino phase defect is modeled by the standard oscillatory phase between mass sectors $i,j$:
\begin{equation}
\Delta\phi_{ij}^{(\nu)}=\frac{\Delta m_{ij}^2L}{2E}\qquad (\hbar=c=1),
\end{equation}

\begin{equation}
\boxed{\epsilon_\nu^{ij}=\frac{1}{2\pi}\Delta\phi_{ij}^{(\nu)}=\frac{\Delta m_{ij}^2L}{4\pi E}.}

For multiple channels one may use
\begin{equation}
\epsilon_\nu=\sum_{i<j}w_{ij}\frac{\Delta m_{ij}^2L}{4\pi E},

\boxed{\Phi_{\rm AB}+\Phi_B+\Phi_E+\Theta_{\mathcal I}=2\pi\left(m+D_\Lambda+\epsilon_\nu\right),\qquad m\in\mathbb Z.}

\section{Relational Lagrangian with phase as the primary free sector}
Let $q^a$ be coordinates on the relational/Kahler base.  Let $\chi\in U(1)$ be the phase coordinate.  Define the covariant phase velocity
\begin{equation}
\boxed{D_t\chi=\dot\chi+\mathcal A_a(q)\dot q^a,}
\end{equation}
where
\begin{equation}
\mathcal A=\mathcal A_{\rm ABE}+\mathcal A_{\mathcal I}.
\end{equation}
A minimal relational Lagrangian is
\begin{equation}
\boxed{L=\frac12g_{ab}(q)\dot q^a\dot q^b+\frac{I_\phi}{2}(D_t\chi)^2+J_0D_t\chi-V(q).}
\end{equation}
The phase momentum is
\begin{equation}
J=\frac{\partial L}{\partial\dot\chi}=I_\phi D_t\chi+J_0,
\end{equation}
so
\begin{equation}
D_t\chi=\frac{J-J_0}{I_\phi}.
\end{equation}
The relational momentum is
\begin{equation}
p_a=\frac{\partial L}{\partial\dot q^a}=g_{ab}\dot q^b+J\mathcal A_a.
\end{equation}
Thus the covariant momentum is
\begin{equation}
\boxed{\Pi_a=p_a-J\mathcal A_a.}
\end{equation}
Solving for $\dot q^a$ gives
\begin{equation}
\dot q^a=g^{ab}\Pi_b.
\end{equation}
The Hamiltonian is
\begin{align}
H&=p_a\dot q^a+J\dot\chi-L\\
&=\frac{(J-J_0)^2}{2I_\phi}+\frac12g^{ab}(p_a-J\mathcal A_a)(p_b-J\mathcal A_b)+V(q).
\end{align}
Therefore
\begin{equation}
\boxed{H=H_\Phi+H_\perp,}
\end{equation}
where
\begin{equation}
\boxed{H_\Phi=\frac{(J-J_0)^2}{2I_\phi},\qquad H_\perp=\frac12g^{ab}\Pi_a\Pi_b+V(q).}
\end{equation}
This is the classical phase-first split.
